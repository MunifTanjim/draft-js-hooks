# Getting Started

## Installation

**npm**:

```sh
npm install @draft-js-hooks/editor react react-dom
```

**yarn**:

```sh
yarn add @draft-js-hooks/editor react react-dom
```

## Usage

```js
import { Editor } from '@draft-js-hooks/editor'
import { EditorState } from 'draft-js'
import 'draft-js/dist/Draft.css'
import React, { useCallback, useRef, useState } from 'react'
import ReactDOM from 'react-dom'

const hooks = []

function HookedEditor() {
  const store = useRef(null)
  const [editorState, setEditorState] = useState(EditorState.createEmpty())

  const onClick = useCallback(() => {
    store.current.getEditor().focus()
  }, [])

  const onChange = useCallback(editorState => {
    setEditorState(editorState)
  }, [])

  return (
    <div onClick={onClick}>
      <Editor
        store={store}
        editorState={editorState}
        onChange={onChange}
        hooks={hooks}
      />
    </div>
  )
}

ReactDOM.render(<HookedEditor />, document.getElementById('container'))
```
